\newglossaryentry{glos:ki}{
  name={Künstliche Intelligenz},
  description={Nachbildung von menschlicher Intelligenz durch Maschinen}
}

\newglossaryentry{glos:llm}{
  name={Large Language Model},
  description={Ein künstliches neuronales Netzwerkmodell, das für die Sprachverarbeitung trainiert wurde und Aufgaben wie Textgenerierung, Übersetzung oder Zusammenfassung beherrscht. Beispiele sind GPT-3 oder GPT-4.}
}

\newglossaryentry{glos:nn}{
  name={Neuronale Netze},
  description={So ein hirn ding... bestimmt schlauer als ich}
}

\newglossaryentry{glos:prompt}{
  name={Prompt},
  description={Texti für Chatty}
}

\newglossaryentry{KI}{
    name={KI},
    description={Ein KI-System ist ein informationsverarbeitendes Computersystem, das mithilfe nicht-strikt-deterministischer Modelle oder Algorithmen automatisiert Eingabedaten analysiert und daraus Ausgaben erzeugt, um komplexe Aufgaben wie Prognosen, Klassifikationen oder Generierungen zu unterstützen. Wird im Folgenden synonym mit LLM verwendet}
}


\newglossaryentry{neuronales-netz}{
    name={Neuronales Netz},
    description={Ein aus mehreren Schichten von Neuronen bestehendes Netzwerk, welches dem Gehirn nachempfunden ist.}
}

\newglossaryentry{large-language-model}{
    name={Large Language Model},
    description={Ein KI-Modell, welches mit einer Vielzahl von Daten trainiert wurde und Text generieren kann.}
}

\newglossaryentry{prompt}{
    name={Prompt},
    description={Text, welcher einem LLM als Eingabe dient.}
}

\newglossaryentry{open-source}{
    name={Open Source},
    description={Software, deren Quellcode öffentlich zugänglich ist.}
}

\newglossaryentry{api}{
    name={API},
    description={Application Programming Interface, Schnittstelle zwischen zwei Anwendungen, die es ihnen erlaubt, miteinander zu kommunizieren.}
}

\newglossaryentry{parameter}{
    name={Parameter},
    description={Hier: Verstellbare Gewichtungen eines NNs oder LLMs, welche durch das Training festgelegt werden.}
}

\newglossaryentry{chatgpt}{
    name={ChatGPT},
    description={Ein LLM von der Firma OpenAI.}
}

\newglossaryentry{backpropagation}{
    name={Backpropagation},
    description={Eine Trainingsmethode für NNs, bei der die Gewichte graduell anhand der Ergebnisse angepasst werden.}
}

\newglossaryentry{transformer}{
    name={Transformer},
    description={Die Transformer-Architektur bezeichnet die Architektur einer KI, welche eine Eingabe in einem bestimmten Format in eine Ausgabe eines anderen Formates umwandelt.}
}

\newglossaryentry{token}{
    name={Token},
    description={Tokens sind die kleinstmöglichen Textbausteine, welche ein LLM verarbeiten kann.}
}

\newglossaryentry{halluzinationen}{
    name={Halluzinationen},
    description={Aussagen eines LLMs, welche plausibel klingen, aber inkorrekt sind (Quelle: Liebert).}
}

\newglossaryentry{self-efficacy}{
    name={Self-Efficacy},
    description={Selbstwirksamkeit, das Vertrauen in die eigenen Fähigkeiten, eine Aufgabe erfolgreich abzuschließen (Self-Efficacy).}
}

\newglossaryentry{ollama}{
    name={Ollama},
    description={Ollama ist eine lokal laufende Plattform für LLMs, welche man vom Terminal aus steuern kann.}
}

\newglossaryentry{tooltips}{
    name={Tooltips},
    description={Ein kleines Pop-Up, welches erscheint, wenn der Mauszeiger über ein Element bewegt wird oder es angeklickt wird.}
}

\newglossaryentry{augmented-intelligence}{
    name={Augmented Intelligence},
    description={Ein KI-basiertes Hilfsmittel, welches nur unterstützend wirkt, jedoch keine Verantwortung (in dem Fall das alleinige Verfassen eines Textes) übernimmt.}
}

\newglossaryentry{benchmarken}{
    name={Benchmarken},
    description={Standardisierte Methoden zum Vergleich verschiedener LLMs.}
}

\newglossaryentry{prompt-engineering}{
    name={Prompt Engineering},
    description={Das gezielte Verändern von KI-Eingaben, um das Ergebnis dem gewünschten anzupassen.}
}
